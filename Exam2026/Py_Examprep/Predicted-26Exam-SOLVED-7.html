# Exam-specific tuning emphasis.
exam_id <- 7
alpha_main <- 1      # 0=ridge, 1=lasso, 0.5=elastic net
cost_fn <- 4        # Relative business cost of false negative

# Helper to compute AUC from ranks (Mann-Whitney interpretation).
auc_rank <- function(y_true, score, positive = "Yes") {
  y_chr <- as.character(y_true)
  y_bin <- if_else(y_chr == positive, 1, 0)
  n1 <- sum(y_bin == 1)
  n0 <- sum(y_bin == 0)
  if (n1 == 0 || n0 == 0) return(NA_real_)
  r <- rank(score)
  (sum(r[y_bin == 1]) - n1 * (n1 + 1) / 2) / (n1 * n0)
}

# Helper for confusion-based metrics from probabilities.
class_metrics <- function(y_true, p_hat, threshold = 0.5, positive = "Yes") {
  y_chr <- as.character(y_true)
  negative <- setdiff(sort(unique(y_chr)), positive)[1]
  actual <- factor(y_chr, levels = c(negative, positive))
  pred <- factor(if_else(p_hat >= threshold, positive, negative), levels = c(negative, positive))
  cm <- table(actual = actual, pred = pred)
  tn <- cm[negative, negative]
  fp <- cm[negative, positive]
  fn <- cm[positive, negative]
  tp <- cm[positive, positive]
  tibble(
    threshold = threshold,
    accuracy = (tp + tn) / sum(cm),
    sensitivity = tp / (tp + fn),
    specificity = tn / (tn + fp),
    auc = auc_rank(actual, p_hat, positive = positive),
    expected_cost = cost_fn * fn + fp
  )
}

# Generic elastic-net objective used to discuss ridge/lasso tradeoff.
objective <- function(beta, X, y, lambda, alpha) {
  rss <- sum((y - X %*% beta)^2)
  pen <- lambda * ((1 - alpha) * sum(beta^2) + alpha * sum(abs(beta)))
  rss + pen
}

Method interpretation for this solved exam: with alpha_main = 1, the focus is LASSO.

(d) Random forest tuning and variable importance

# Cross-validated mtry search.
set.seed(90 + exam_id)
fold_id <- sample(rep(1:5, length.out = nrow(train_c)))
mtry_grid <- c(2, 4, 6, 8)

rf_cv <- map_dfr(mtry_grid, function(m) {
  aucs <- map_dbl(1:5, function(k) {
    tr <- train_c[fold_id != k, ]
    va <- train_c[fold_id == k, ]
    fit_k <- randomForest(churn ~ . - state, data = tr, ntree = 400, mtry = m)
    p_k <- predict(fit_k, newdata = va, type = "prob")[, "Yes"]
    auc_rank(va$churn, p_k)
  })
  tibble(mtry = m, cv_auc = mean(aucs))
})

best_mtry <- rf_cv %>% slice_max(cv_auc, n = 1) %>% pull(mtry)
rf_fit <- randomForest(churn ~ . - state, data = train_c, ntree = 500, mtry = best_mtry, importance = TRUE)
p_rf <- predict(rf_fit, newdata = test_c, type = "prob")[, "Yes"]
rf_eval <- class_metrics(test_c$churn, p_rf, threshold = best_thr$threshold)

(e) Boosting and final recommendation

# Prepare numeric target for gbm Bernoulli.
train_g <- train_c %>% mutate(churn01 = if_else(churn == "Yes", 1, 0))
test_g <- test_c %>% mutate(churn01 = if_else(churn == "Yes", 1, 0))

set.seed(140 + exam_id)
gbm_fit <- gbm(
  churn01 ~ . - churn - state,
  distribution = "bernoulli",
  data = train_g,
  n.trees = 600,
  interaction.depth = 3,
  shrinkage = 0.03,
  n.minobsinnode = 15,
  verbose = FALSE
)

# Stage-wise AUC to choose tree count.
tree_grid <- seq(25, 600, by = 25)
auc_path <- map_dbl(tree_grid, function(nt) {
  p_nt <- predict(gbm_fit, newdata = test_g, n.trees = nt, type = "response")
  auc_rank(test_c$churn, p_nt)
})
best_trees <- tree_grid[which.max(auc_path)]

p_boost <- predict(gbm_fit, newdata = test_g, n.trees = best_trees, type = "response")
boost_eval <- class_metrics(test_c$churn, p_boost, threshold = best_thr$threshold)

comparison <- bind_rows(
  best_thr %>% mutate(model = "Logistic"),
  rf_eval %>% mutate(model = paste0("Random Forest (mtry=", best_mtry, ")")),
